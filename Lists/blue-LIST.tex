%%%%%
%%
%% This file sets up the Blue datatype and creates Blue macros.  These
%% are for bluesheets.
%%
%%%%%

\DECLARESUBTYPE{Blue}{Element}
\PRESETS{Blue}{
  \FD\MYfile	{} %% filename (in \blues directory)
  \FS\MYtext	{\ifx\MYfile\empty\else%
		  \getextractenvs{document}{\blues/\MYfile}%
		\fi}
  }


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\NEW{Blue}{\bTest}{
  \s\MYname	{Test Bluesheet}
  \s\MYfile	{README.tex}
}

  \s\MYfile	{Space.tex}
}
